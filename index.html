async function load() { const matches = await fetchData("MATCHES_PLAYED"); const tables = await fetchData("TABLES_SORTED"); const semis = await fetchData("SEMIS"); const finals = await fetchData("FINALS"); document.getElementById("matches").innerHTML = matches.length ? matches.map(m => `
${Object.values(m).join(" | ")}
`).join("") : "No completed matches"; document.getElementById("tables").innerHTML = tables.length ? tables.map(t => `
${Object.values(t).join(" | ")}
`).join("") : "No table data"; document.getElementById("semis").innerHTML = semis.length ? semis.map(s => `
${Object.values(s).join(" | ")}
`).join("") : "TBD"; document.getElementById("finals").innerHTML = finals.length ? finals.map(f => `
${Object.values(f).join(" | ")}
`).join("") : "TBD"; } load();
